    <script>
        let myChart;
        let currentProfile = 'conservative';
        const profiles = {
            conservative: { rates: , labels:, color: ['#10b981', '#3b82f6', '#94a3b8'], text: "Focus on wealth preservation. 50% in Total Bond Market <span class='ticker'>BND</span>." },
            moderate: { rates: , labels:, color: ['#10b981', '#3b82f6', '#94a3b8'], text: "Balanced growth. 60% in S&P 500 Index <span class='ticker'>VOO</span>." },
            aggressive: { rates: , labels:, color: ['#f59e0b', '#3b82f6', '#10b981'], text: "High growth potential. 90% in Nasdaq-100 <span class='ticker'>QQQ</span>." }
        };

        function setProfile(p, btn) {
            currentProfile = p;
            document.querySelectorAll('button').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            updateStrategy();
        }

        function updateStrategy() {
            const amount = parseFloat(document.getElementById('amount').value) |

| 0;
            const p = profiles[currentProfile];
            const values = p.rates.map(r => (amount * r / 100).toFixed(2));
            
            document.getElementById('strategy-text').innerHTML = `
                <strong>Plan for $${amount.toLocaleString()}:</strong><br>
                ${p.text}<br><br>
                • $${values} to ${p.labels}<br>
                • $${values[1]} to ${p.labels[1]}<br>
                • $${values[2]} to ${p.labels[2]}
            `;

            if (myChart) myChart.destroy();
            const ctx = document.getElementById('allocationChart').getContext('2d');
            myChart = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: p.labels,
                    datasets:
                },
                options: { cutout: '70%', plugins: { legend: { position: 'bottom' } } }
            });
        }
        updateStrategy();
    </script>
